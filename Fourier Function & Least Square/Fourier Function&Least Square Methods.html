def FT(x):
    x = np.asarray(x, dtype=float)
    N = x.shape[0]
    n = np.arange(N)
    k = n.reshape((N, 1))
    M = np.exp(-2j * np.pi * k * n / N)
    M = M / len(frounded)
    return np.dot(M, x)

ft = FT(frounded)
ft = np.abs(ft)



def show_ft():
    y = np.arange(1, 502)
    plt.plot(ft)
    plt.title('  ')
    plt.grid()
    plt.show()
show_ft()
